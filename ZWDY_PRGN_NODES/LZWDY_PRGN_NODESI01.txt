*       text
*----------------------------------------------------------------------*
MODULE USER_COMMAND_0100 INPUT.
  IF RB_INT = 'X'.
    MESSAGE W001(00) WITH text-010. "#EC MG_PAR_CNT
  ENDIF.

  CASE ok_code.
    WHEN 'OK'.
      if wdy_name is initial.
        message e002.
      else.
        if wdy_text is initial.
          perform get_wda_text_new changing wdy_text.
        endif.
        PERFORM check_wdy.
        CLEAR ok_code.
        SET SCREEN 0.
      endif.
    WHEN 'SDEL'.
      PERFORM delete_param_line.
      CLEAR ok_code.

  ENDCASE.
ENDMODULE.                 " USER_COMMAND_0100  INPUT

*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE PARAM_CNTRL_MODIFY INPUT.
  CHECK ok_code NE 'SDEL'.
  CHECK param_lin IS NOT INITIAL.
  PERFORM check_param USING param_lin
                      CHANGING new_entry.
  IF new_entry = 'X'.
    APPEND param_lin TO i_param.
  ELSE.
    MODIFY i_param
      FROM param_lin
      INDEX param_cntrl-current_line.
  ENDIF.

ENDMODULE.                 " PARAM_CNTRL_MODIFY  INPUT
